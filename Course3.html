get_streak = function(x){
  y = rep(0,length(x))
  y[x == "H"] = 1
  y = c(0, y, 0)
  wz = which(y == 0)
  streak = diff(wz) - 1
  return(streak)
}

    sample_means10[i] = mean(samp)
    samp = sample(area, 50)
    sample_means50[i] = mean(samp)
    samp = sample(area, 100)

}
